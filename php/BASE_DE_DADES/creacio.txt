//CREAR BASE DE DADES

CREATE DATABASE EP_PRA1;

//CREAR TAULES

CREATE TABLE HABITACIONS (
    codi INT PRIMARY KEY,
    tipus ENUM('SIMPLE','DOBLE','SUITE') NOT NULL,
    preuNit INT,
    descripcio VARCHAR(100)
);

CREATE TABLE USUARIS (
    DNI INT PRIMARY KEY,
    contrasenya VARCHAR(20) NOT NULL,
    estat ENUM('NO AUTENTICAT','AUTENTICAT','BLOQUEJAT'),
    numErrors INT,
    tipus ENUM('ADMINISTRADOR','CLIENT')
);

CREATE TABLE CLIENTS (
    DNI INT PRIMARY KEY,
    nom VARCHAR(20) NOT NULL,
    adreça VARCHAR(20) NOT NULL,
    telefon INT,
    email VARCHAR(20),
    fotografia VARCHAR(100),
    FOREIGN KEY (DNI) REFERENCES USUARIS(DNI)
);

CREATE TABLE RESERVES (
    numReserva INT PRIMARY KEY,
    codiHabitacio INT,
    dniCliente INT,
    dataEntrada DATE,
    dataSortida DATE,
    preuTotal INT,
    estat ENUM('RESERVADA','OCUPADA','REALITZADA'),
    FOREIGN KEY (codiHabitacio) REFERENCES HABITACIONS(codi),
    FOREIGN KEY (dniCliente) REFERENCES CLIENTS(DNI)
);
